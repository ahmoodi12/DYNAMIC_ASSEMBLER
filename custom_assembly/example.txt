# python3 assembler_v1.py "custom assembly/example.txt" "custom assembly/example.json"

.def x, r0
.def y, r1
.def z, r2
.def max, r3

.start $0x10 
ldi y, 1   
ldi max, 15        

loop:
    add x, y, z    
    mov x, y        
    mov y, z       
    cmp max, x    
    jnn loop      

__end:
    hlt
